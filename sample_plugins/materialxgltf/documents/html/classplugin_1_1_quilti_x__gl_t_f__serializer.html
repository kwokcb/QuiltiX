272 def setup_default_export_options(self, path, bakeFileName, bakeResolution=1024, embed_geometry=False) -> dict:
273 '''
274 Set up the default export options for gltf output.
275 @param path (str): path to the gltf file
276 @param bakeFileName (str): path to the baked file
277 @param bakeResolution (int): resolution of the baked textures. Default is 1024.
278 @param embed_geometry (bool): whether to embed the geometry in the gltf file. Default is False.
279 @return options (dict): Dictionary of options for the conversion
280 '''
281 options = core.MTLX2GLTFOptions()
282
283 options['debugOutput'] = False
284 options['bakeFileName'] = bakeFileName
285 # Clamp to a minimum power-of-2 resolution
286 bakeResolution = max(bakeResolution, 16)
287 options['bakeResolution'] = bakeResolution
288
289 # TODO: Expose these options to the user
290 if embed_geometry:
291 # By default uses the "MaterialX" shader ball provided as part of
292 # the materialxgltf package for binary export.

294 msg = '> Load default geometry: %s' % mx.FilePath(gltfGeometryFile).getBaseName()
295 logger.info(msg)
296 options['geometryFile'] = gltfGeometryFile
297 options['primsPerMaterial'] = True
298 options['writeDefaultInputs'] = False
299 options['translateShaders'] = True
300 options['bakeTextures'] = True
301 # Always should be set
302 options['addExtractNodes'] = True
303
304 searchPath = mx.getDefaultDataSearchPath()
305 if not mx.FilePath(path).isAbsolute():
306 path = os.path.abspath(path)
307 searchPath.append(mx.FilePath(path).getParentPath())
308 searchPath.append(mx.FilePath.getCurrentPath())
309 if embed_geometry:
