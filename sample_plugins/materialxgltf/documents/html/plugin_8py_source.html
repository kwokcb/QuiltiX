286 def import_gltf_triggered(self) -> None:
287 '''
288 Import a glTF file into the current graph.
289 '''
290 start_path = self.editor.mx_selection_path
291 if not start_path:
292 start_path = self.editor.geometry_selection_path
293
294 if not start_path:
295 start_path = os.path.join(ROOT, "resources", "materials")
296
297 path = self.editor.request_filepath(
298 title="Load glTF file", start_path=start_path, file_filter="glTF files (*.gltf)", mode="open",
299 )
300
301 if not os.path.exists(path):
302 logger.error('Cannot find input file: ' + path)
303 return
304
305 # Setup defaults for glTF import and create the MaterialX document
306 # - Do not create assignments
307 # - Do not add all inputs
308 # - Add extract nodes. Only required for pre 1.39 MaterialX
309 options = core.GLTF2MtlxOptions()
310 options['createAssignments'] = False
311 options['addAllInputs'] = False
312 options['addExtractNodes'] = True
313 gltf2MtlxReader = core.GLTF2MtlxReader()
314 gltf2MtlxReader.setOptions(options)
315 doc = gltf2MtlxReader.convert(path)
316
317 success = True
318 if not doc:
319 success = False
320 logger.error('Error converting glTF file to MaterialX file')
321 else:
322 success , err = doc.validate()
323 if not success:
324 logger.error(err)
325 else:
326 # Extract out the appropriate data from the original MaterialX document
327 docString = core.Util.writeMaterialXDocString(doc)
328 doc = mx.createDocument()
329 mx.readFromXmlString(doc, docString)
330
331 # Auto ng creation can cause issues for nodes with multiple outputs. Turn off on import
332 ng_abstraction = self.editor.act_ng_abstraction.isChecked()
333 self.editor.act_ng_abstraction.setChecked(False)
334
335 self.editor.mx_selection_path = path
336 self.editor.qx_node_graph.load_graph_from_mx_doc(doc)
337 self.editor.qx_node_graph.mx_file_loaded.emit(path)
338
339 self.editor.act_ng_abstraction.setChecked(ng_abstraction)

341 def setup_default_export_options(self, path, bakeFileName, bakeResolution=1024, embed_geometry=False) -> dict:
342 '''
343 Set up the default export options for gltf output.
344 @param path (str): path to the gltf file
345 @param bakeFileName (str): path to the baked file
346 @param bakeResolution (int): resolution of the baked textures. Default is 1024.
347 @param embed_geometry (bool): whether to embed the geometry in the gltf file. Default is False.
348 @return options (dict): Dictionary of options for the conversion
349 '''
350 options = core.MTLX2GLTFOptions()
351
352 options['debugOutput'] = False
353 options['bakeFileName'] = bakeFileName
354 # Clamp to a minimum power-of-2 resolution
355 bakeResolution = max(bakeResolution, 16)
356 options['bakeResolution'] = bakeResolution
357
358 # TODO: Expose these options to the user
359 if embed_geometry:
360 # By default uses the "MaterialX" shader ball provided as part of
361 # the materialxgltf package for binary export.
362 gltfGeometryFile = pkg_resources.resource_filename('materialxgltf', 'data/shaderBall.gltf')
363 msg = '> Load default geometry: %s' % mx.FilePath(gltfGeometryFile).getBaseName()
364 logger.info(msg)
365 options['geometryFile'] = gltfGeometryFile
366 options['primsPerMaterial'] = True
367 options['writeDefaultInputs'] = False
368 options['translateShaders'] = True
369 options['bakeTextures'] = True
370 # Always should be set
371 options['addExtractNodes'] = True
372
373 searchPath = mx.getDefaultDataSearchPath()
374 if not mx.FilePath(path).isAbsolute():
375 path = os.path.abspath(path)
376 searchPath.append(mx.FilePath(path).getParentPath())
377 searchPath.append(mx.FilePath.getCurrentPath())
378 if embed_geometry:
379 searchPath.append(mx.FilePath(gltfGeometryFile).getParentPath())
380 options['searchPath'] = searchPath
381
382 return options

397 def export_gltf_triggered(self) -> None:
398 '''
399 Export the current graph to a glTF file in binary format (glb)
400 - Will perform shader translation if needed to glTF
401 - Will perform baking if needed
402 - Will package to a binary file
403 @return: None
404 '''
405 start_path = self.editor.mx_selection_path
406 if not start_path:
407 start_path = self.editor.geometry_selection_path
408
409 if not start_path:
410 start_path = os.path.join(ROOT, "resources", "materials")
411
412 path = self.editor.request_filepath(
413 title="Save glTF file", start_path=start_path, file_filter="glTF files (*.gltf)", mode="save",
414 )
415
416 # Set up export options
417 bakeFileName = self.create_baked_path(start_path)
418 options = self.setup_default_export_options(path, bakeFileName, bakeResolution=1024, embed_geometry=True)
419 gltf_string = self.convert_graph_to_gltf(options)
420
421 if gltf_string == '{}':
422 return
423
424 self.editor.set_current_filepath(path)
425
426 # To be able to view the exported file in the glTF viewer, we need to package to a binary file
427 options['packageBinary'] = True
428 try:
429 with open(path, "w") as f:
430 f.write(gltf_string)
431 logger.info(f"Wrote .gltf file to {path}")
432
433 # Package binary file
434 binaryFileName = str(path)
435 binaryFileName = binaryFileName.replace('.gltf', '.glb')
436 logger.debug('- Packaging GLB file...')
437 mtlx2glTFWriter = core.MTLX2GLTFWriter()
438 mtlx2glTFWriter.setOptions(options)
439 saved, images, buffers = mtlx2glTFWriter.packageGLTF(path, binaryFileName)
440 logger.info('- Save GLB file:' + binaryFileName + '. Status:' + str(saved))
441 for image in images:
442 logger.debug(' - Embedded image: ' + image)
443 for buffer in buffers:
444 logger.debug(' - Embedded buffer: ' + buffer)
445 logger.debug('- Packaging GLB file... finished.')
446 except Exception as e:
447 logger.error(e)

450 def convert_graph_to_gltf(self, options) -> str:
451 '''
452 Convert the current graph to a glTF document string.
453 Will perform:
454 - Shader translation if needed (not that only standard surface is supported)
455 - Baking if needed. Note that this writes local files.
456 - Uses the materialgltf package to perform conversion
457
458 @param options (dict): Dictionary of options for the conversion
459 @return The glTF string.
460 '''
461 # Disable auto nodegraph creation during
462 ng_abstraction = self.editor.act_ng_abstraction.isChecked()
463 self.editor.act_ng_abstraction.setChecked(False)
464
465 doc = self.editor.qx_node_graph.get_current_mx_graph_doc()
466
467 self.editor.act_ng_abstraction.setChecked(ng_abstraction)
468
469 mtlx2glTFWriter = core.MTLX2GLTFWriter()
470 mtlx2glTFWriter.setOptions(options)
471
472 # Need to load in definition libraries to get translation graphs
473 stdlib = mx.createDocument()
474 searchPath = mx.getDefaultDataSearchPath()
475 libraryFolders = []
476 libraryFolders.extend(mx.getDefaultDataLibraryFolders())
477 mx.loadLibraries(libraryFolders, searchPath, stdlib)
478 doc.importLibrary(stdlib)
479
480 # Perform shader translation if needed
481 translatedCount = 0
482 if options['translateShaders']:
483 translatedCount = mtlx2glTFWriter.translateShaders(doc)
484 logger.debug('- Translated shaders: ' + str(translatedCount))
485
486 forceBake = False
487 if self.bake_textures_option.isChecked():
488 logger.debug('--- Forcing baking of textures')
489 forceBake = True
490
491 # Perform baking if needed
492 if forceBake or (translatedCount > 0 and options['bakeTextures']):
493 bakedFileName = options['bakeFileName']
494 bakeResolution = 1024
495 if options['bakeResolution']:
496 bakeResolution = options['bakeResolution']
497 logger.debug(f'- START baking to {bakedFileName}. Resolution: {bakeResolution} ...')
498 mtlx2glTFWriter.bakeTextures(doc, False, bakeResolution, bakeResolution, False,
499 False, False, bakedFileName)
500 if os.path.exists(bakedFileName):
501 logger.debug(' - Baked textures to: ' + bakedFileName)
502 doc, libFiles = core.Util.createMaterialXDoc()
503 mx.readFromXmlFile(doc, bakedFileName, options['searchPath'])
504 remappedUris = core.Util.makeFilePathsRelative(doc, bakedFileName)
505 for uri in remappedUris:
506 logger.debug(' - Remapped URI: ' + uri[0] + ' to ' + uri[1])
507 core.Util.writeMaterialXDoc(doc, bakedFileName)
508 logger.debug('- ... END baking.')
509
510 gltfString = mtlx2glTFWriter.convert(doc)
511 return gltfString
