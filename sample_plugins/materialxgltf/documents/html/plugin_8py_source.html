217  def import_gltf_triggered(self) -> None:
218  '''
219  Import a glTF file into the current graph.
220  '''
221  start_path = self.editoreditor.mx_selection_path
222  if not start_path:
223  start_path = self.editoreditor.geometry_selection_path
224 
225  if not start_path:
226  start_path = os.path.join(ROOT, "resources", "materials")
227 
228  path = self.editoreditor.request_filepath(
229  title="Load glTF file", start_path=start_path, file_filter="glTF files (*.gltf)", mode="open",
230  )
231 
232  if not os.path.exists(path):
233  logger.error('Cannot find input file: ' + path)
234  return
235 
236  # Setup defaults for glTF import and create the MaterialX document
237  # - Do not create assignments
238  # - Do not add all inputs
239  # - Add extract nodes. Only required for pre 1.39 MaterialX
240  options = core.GLTF2MtlxOptions()
241  options['createAssignments'] = False
242  options['addAllInputs'] = False
243  options['addExtractNodes'] = True
244  gltf2MtlxReader = core.GLTF2MtlxReader()
245  gltf2MtlxReader.setOptions(options)
246  doc = gltf2MtlxReader.convert(path)
247 
248  success = True
249  if not doc:
250  success = False
251  logger.error('Error converting glTF file to MaterialX file')
252  else:
253  success , err = doc.validate()
254  if not success:
255  logger.error(err)
256  else:
257  # Extract out the appropriate data from the original MaterialX document
258  docString = core.Util.writeMaterialXDocString(doc)
259  doc = mx.createDocument()
260  mx.readFromXmlString(doc, docString)
261 
262  # Auto ng creation can cause issues for nodes with multiple outputs. Turn off on import
263  ng_abstraction = self.editoreditor.act_ng_abstraction.isChecked()
264  self.editoreditor.act_ng_abstraction.setChecked(False)
265 
266  self.editoreditor.mx_selection_path = path
267  self.editoreditor.qx_node_graph.load_graph_from_mx_doc(doc)
268  self.editoreditor.qx_node_graph.mx_file_loaded.emit(path)

272  def setup_default_export_options(self, path, bakeFileName, embed_geometry=False) -> dict:
273  '''
274  Set up the default export options for gltf output.
275  @param path (str): path to the gltf file
276  @param bakeFileName (str): path to the baked file
277  @param embed_geometry (bool): whether to embed the geometry in the gltf file. Default is False.
278  @return options (dict): Dictionary of options for the conversion
279  '''
280  options = core.MTLX2GLTFOptions()
281 
282  options['debugOutput'] = False
283  options['bakeFileName'] = bakeFileName
284 
285  # TODO: Expose these options to the user
286  if embed_geometry:
287  # By default uses the "MaterialX" shader ball provided as part of
288  # the materialxgltf package for binary export.
289  gltfGeometryFile = pkg_resources.resource_filename('materialxgltf', 'data/shaderBall.gltf')
290  msg = '> Load default geometry: %s' % mx.FilePath(gltfGeometryFile).getBaseName()
291  logger.info(msg)
292  options['geometryFile'] = gltfGeometryFile
293  options['primsPerMaterial'] = True
294  options['writeDefaultInputs'] = False
295  options['translateShaders'] = True
296  options['bakeTextures'] = True
297  # Always should be set
298  options['addExtractNodes'] = True
299 
300  searchPath = mx.getDefaultDataSearchPath()
301  if not mx.FilePath(path).isAbsolute():
302  path = os.path.abspath(path)
303  searchPath.append(mx.FilePath(path).getParentPath())
304  searchPath.append(mx.FilePath.getCurrentPath())
305  if embed_geometry:
306  searchPath.append(mx.FilePath(gltfGeometryFile).getParentPath())
307  options['searchPath'] = searchPath
308 
309  return options

328  )
329

336  return

343  with open(path, "w") as f:

353  saved, images, buffers = mtlx2glTFWriter.packageGLTF(path, binaryFileName)
354  logger.info('- Save GLB file:' + binaryFileName + '. Status:' + str(saved))
355  for image in images:
356  logger.debug(' - Embedded image: ' + image)

364  def convert_graph_to_gltf(self, options) -> str:
365  '''
366  Convert the current graph to a glTF document string.
367  Will perform:
368  - Shader translation if needed (not that only standard surface is supported)
369  - Baking if needed. Note that this writes local files.
370  - Uses the materialgltf package to perform conversion
371 
372  @param options (dict): Dictionary of options for the conversion
373  @return The glTF string.
374  '''
375  # Disable auto nodegraph creation during
376  ng_abstraction = self.editoreditor.act_ng_abstraction.isChecked()
377  self.editoreditor.act_ng_abstraction.setChecked(False)
378 
379  doc = self.editoreditor.qx_node_graph.get_current_mx_graph_doc()
380 
381  self.editoreditor.act_ng_abstraction.setChecked(ng_abstraction)
382 
383  mtlx2glTFWriter = core.MTLX2GLTFWriter()
384  mtlx2glTFWriter.setOptions(options)
385 
386  # Need to load in definition libraries to get translation graphs
387  stdlib = mx.createDocument()
388  searchPath = mx.getDefaultDataSearchPath()
389  libraryFolders = []
390  libraryFolders.extend(mx.getDefaultDataLibraryFolders())
391  mx.loadLibraries(libraryFolders, searchPath, stdlib)
392  doc.importLibrary(stdlib)
393 
394  # Perform shader translation if needed
395  translatedCount = 0
396  if options['translateShaders']:
397  translatedCount = mtlx2glTFWriter.translateShaders(doc)
398  logger.debug('- Translated shaders: ' + str(translatedCount))
399 
400  forceBake = False
401  if self.bake_textures_optionbake_textures_option.isChecked():
402  logger.debug('--- Forcing baking of textures')
403  forceBake = True
404 
405  # Perform baking if needed
406  if forceBake or (translatedCount > 0 and options['bakeTextures']):
407  logger.debug('- Baking start...')
408  bakeResolution = 2048
409  bakedFileName = options['bakeFileName']
410  if options['bakeResolution']:
411  bakeResolution = options['bakeResolution']
412  mtlx2glTFWriter.bakeTextures(doc, False, bakeResolution, bakeResolution, False,
413  False, False, bakedFileName)
414  if os.path.exists(bakedFileName):
415  logger.debug(' - Baked textures to: ' + bakedFileName)
416  doc, libFiles = core.Util.createMaterialXDoc()
417  mx.readFromXmlFile(doc, bakedFileName, options['searchPath'])
418  remappedUris = core.Util.makeFilePathsRelative(doc, bakedFileName)
419  for uri in remappedUris:
420  logger.debug(' - Remapped URI: ' + uri[0] + ' to ' + uri[1])
421  core.Util.writeMaterialXDoc(doc, bakedFileName)
422  logger.debug('- ...Baking end.')
423 
424  gltfString = mtlx2glTFWriter.convert(doc)
425  return gltfString
